107template <class RobotType, class ObjectType>
108void HighwayRoadMap<RobotType, ObjectType>::search() {
109 // Construct the roadmap
110 const Index num_vtx = res_.graphStructure.vertex.size();
111 AdjGraph g(num_vtx);
112
113 for (Index i = 0; i < res_.graphStructure.edge.size(); ++i) {
114 boost::add_edge(Index(res_.graphStructure.edge[i].first),
115 Index(res_.graphStructure.edge[i].second),
116 Weight(res_.graphStructure.weight[i]), g);
117 }
118
119 // Locate the nearest vertex for start and goal in the roadmap
120 const std::vector<Vertex> idx_s = getNearestNeighborsOnGraph(
121 start_, param_.numSearchNeighbor, param_.searchRadius);
122 const std::vector<Vertex> idx_g = getNearestNeighborsOnGraph(
123 goal_, param_.numSearchNeighbor, param_.searchRadius);
124
125 // Search for shortest path in the searching regions
126 Index num;
127 for (Vertex idxS : idx_s) {
128 for (Vertex idxG : idx_g) {
129 std::vector<Vertex> p(num_vertices(g));
130 std::vector<double> d(num_vertices(g));
131
132 try {
133 boost::astar_search(
134 g, idxS,
135 [this, idxG](Vertex v) {
136 return vectorEuclidean(
137 res_.graphStructure.vertex[v],
138 res_.graphStructure.vertex[idxG]);
139 },
140 boost::predecessor_map(
141 boost::make_iterator_property_map(
142 p.begin(), get(boost::vertex_index, g)))
143 .distance_map(make_iterator_property_map(
144 d.begin(), get(boost::vertex_index, g)))
145 .visitor(AStarGoalVisitor<Vertex>(idxG)));
146 } catch (AStarFoundGoal found) {
147 // Record path and cost
148 num = 0;
149 res_.solutionPath.cost = 0.0;
150 res_.solutionPath.PathId.push_back(int(idxG));
151 while (res_.solutionPath.PathId[num] != int(idxS) &&
152 num <= num_vtx) {
153 res_.solutionPath.PathId.push_back(
154 int(p[size_t(res_.solutionPath.PathId[num])]));
155 res_.solutionPath.cost +=
156 res_.graphStructure
157 .weight[size_t(res_.solutionPath.PathId[num])];
158 num++;
159 }
160 std::reverse(std::begin(res_.solutionPath.PathId),
161 std::end(res_.solutionPath.PathId));
162
163 if (num == num_vtx + 1) {
164 res_.solutionPath.PathId.clear();
165 res_.solutionPath.cost = INFINITY;
166 } else {
167 res_.solved = true;
168 return;
169 }
170 }
171 }
172 }

175template <class RobotType, class ObjectType>
176void HighwayRoadMap<RobotType, ObjectType>::refineExistRoadmap(
177 const double timeLim) {
178 isRefine_ = true;
179
180 vertexIdxAll_.push_back(vertexIdx_);
181 vertexIdx_.clear();
182
183 param_.numLineX *= 2;
184 param_.numLineY *= 2;
185
186 for (size_t i = 0; i < param_.numSlice; ++i) {
187 auto start = Clock::now();
188
189 // construct refined C-slice
190 constructOneSlice(i);
191 numVertex_.slice = res_.graphStructure.vertex.size();
192 vertexIdx_.push_back(numVertex_);
193
194 // Connect with existing slices
195 connectExistSlice(i);
196
197 res_.planningTime.buildTime += Durationd(Clock::now() - start).count();
198
199 // Search
200 start = Clock::now();
201 search();
202 res_.planningTime.searchTime += Durationd(Clock::now() - start).count();
203
204 res_.planningTime.totalTime =
205 res_.planningTime.buildTime + res_.planningTime.searchTime;
206
207 if (res_.solved || res_.planningTime.totalTime > timeLim) {
208 return;
209 }
210 }
211
212 isRefine_ = false;

215template <class RobotType, class ObjectType>
216void HighwayRoadMap<RobotType, ObjectType>::connectOneSlice2D(
217 const FreeSegment2D& freeSeg) {
218 // Add connections to edge list
219 Index n1 = 0;
220 Index n2 = 0;
221
222 for (size_t i = 0; i < freeSeg.ty.size(); ++i) {
223 n1 = numVertex_.plane.at(i);
224
225 for (size_t j1 = 0; j1 < freeSeg.xM[i].size(); ++j1) {
226 // Connect vertex within the same sweep line
227 if (j1 != freeSeg.xM[i].size() - 1) {
228 if (std::fabs(freeSeg.xU[i][j1] - freeSeg.xL[i][j1 + 1]) <
229 1e-6) {
230 res_.graphStructure.edge.push_back(
231 std::make_pair(n1 + j1, n1 + j1 + 1));
232 res_.graphStructure.weight.push_back(vectorEuclidean(
233 res_.graphStructure.vertex[n1 + j1],
234 res_.graphStructure.vertex[n1 + j1 + 1]));
235 }
236 }
237
238 // Connect vertex btw adjacent sweep lines
239 if (i != freeSeg.ty.size() - 1) {
240 n2 = numVertex_.plane.at(i + 1);
241
242 for (Index j2 = 0; j2 < freeSeg.xM[i + 1].size(); ++j2) {
243 if (isSameSliceTransitionFree(
244 res_.graphStructure.vertex[n1 + j1],
245 res_.graphStructure.vertex[n2 + j2])) {
246 // Direct success connection
247 res_.graphStructure.edge.push_back(
248 std::make_pair(n1 + j1, n2 + j2));
249 res_.graphStructure.weight.push_back(vectorEuclidean(
250 res_.graphStructure.vertex[n1 + j1],
251 res_.graphStructure.vertex[n2 + j2]));
252 } else {
253 bridgeVertex(n1 + j1, n2 + j2);
254 }
255 }
256 }
257 }
258 }

261template <class RobotType, class ObjectType>
262std::vector<std::vector<Coordinate>>
263HighwayRoadMap<RobotType, ObjectType>::getSolutionPath() {
264 std::vector<std::vector<Coordinate>> path;
265 auto poseSize = res_.graphStructure.vertex.at(0).size();
266
267 // Start pose
268 start_.resize(poseSize);
269 path.push_back(start_);
270
271 // Iteratively store intermediate poses along the solved path
272 for (auto pathId : res_.solutionPath.PathId) {
273 path.push_back(res_.graphStructure.vertex.at(size_t(pathId)));
274 }
275
276 // Goal pose
277 goal_.resize(poseSize);
278 path.push_back(goal_);
279
280 return path;
281}

292template <class RobotType, class ObjectType>
293void HighwayRoadMap<RobotType, ObjectType>::bridgeVertex(const Index idx1,
294 const Index idx2) {
295 const auto v1 = res_.graphStructure.vertex.at(idx1);
296 const auto v2 = res_.graphStructure.vertex.at(idx2);
297
298 // Generate new bridge vertex
299 auto vNew1 = v1;
300 vNew1.at(2) = v2.at(2);
301 auto vNew2 = v2;
302 vNew2.at(2) = v1.at(2);
303
304 auto vNew = v1;
305
306 // Check validity of potential connections
307 if (isSameSliceTransitionFree(v1, vNew1) &&
308 isSameSliceTransitionFree(vNew1, v2)) {
309 vNew = vNew1;
310 } else if (isSameSliceTransitionFree(v1, vNew2) &&
311 isSameSliceTransitionFree(vNew2, v2)) {
312 vNew = vNew2;
313 } else {
314 return;
315 }
316
317 // Add new bridge vertex to graph is new connection is valid
318 int idxNew = res_.graphStructure.vertex.size();
319 res_.graphStructure.vertex.push_back(vNew);
320 res_.graphStructure.edge.push_back(std::make_pair(idx1, idxNew));
321 res_.graphStructure.weight.push_back(vectorEuclidean(v1, vNew));
322 res_.graphStructure.edge.push_back(std::make_pair(idxNew, idx2));
323 res_.graphStructure.weight.push_back(vectorEuclidean(vNew, v2));
324}
