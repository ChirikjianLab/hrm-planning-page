63 segment_.xL.clear();
64 segment_.xU.clear();
65 segment_.xM.clear();
66 std::vector<Interval> interval[ty.size()];
67
68 // y-coord
69 segment_.ty = ty;
70
71 // Collision-free line segment for each ty
72 for (Index i = 0; i < ty.size(); ++i) {
73 // Construct intervals at each sweep line
74 const auto lineIdx = static_cast<Eigen::Index>(i);
75 interval[i] = computeSweepLineFreeSegment(lineIdx);
76
77 // x-(z-)coords
78 std::vector<Coordinate> xL;
79 std::vector<Coordinate> xU;
80 std::vector<Coordinate> xM;
81
82 for (Index j = 0; j < interval[i].size(); ++j) {
83 xL.push_back(interval[i][j].s());
84 xU.push_back(interval[i][j].e());
85 xM.push_back((interval[i][j].s() + interval[i][j].e()) / 2.0);
86 }
87 segment_.xL.push_back(xL);
88 segment_.xU.push_back(xU);
89 segment_.xM.push_back(xM);
90 }

97template <typename RobotType, typename ObjectType>
98std::vector<Interval>
99FreeSpaceComputator<RobotType, ObjectType>::computeSweepLineFreeSegment(
100 const Eigen::Index& lineIdx) {
101 // Construct intervals of the segment_ sweep line
102 std::vector<Interval> obsSegment;
103 std::vector<Interval> arenaSegment;
104
105 // Remove NaN terms
106 for (auto j = 0; j < intersect_.arenaLow.at(lineIdx).size(); ++j) {
107 if (!std::isnan(intersect_.arenaLow.at(lineIdx).at(j)) &&
108 !std::isnan(intersect_.arenaUpp.at(lineIdx).at(j))) {
109 arenaSegment.push_back({intersect_.arenaLow.at(lineIdx).at(j),
110 intersect_.arenaUpp.at(lineIdx).at(j)});
111 }
112 }
113 for (auto j = 0; j < intersect_.obstacleLow.at(lineIdx).size(); ++j) {
114 if (!std::isnan(intersect_.obstacleLow.at(lineIdx).at(j)) &&
115 !std::isnan(intersect_.obstacleUpp.at(lineIdx).at(j))) {
116 obsSegment.push_back({intersect_.obstacleLow.at(lineIdx).at(j),
117 intersect_.obstacleUpp.at(lineIdx).at(j)});
118 }
119 }
120
121 // Set operations for Collision-free intervals at each line
122 // Collision-free intervals at each line
123 const std::vector<Interval> collisionFreeSegment = Interval::complements(
124 Interval::intersects(arenaSegment), Interval::unions(obsSegment));
125
126 return collisionFreeSegment;

129template <typename RobotType, typename ObjectType>
130void FreeSpaceComputator<RobotType, ObjectType>::enhanceFreeSegment() {
131 FreeSegment2D enhanced = segment_;
132
133 // Add new vertices within on sweep line
134 for (size_t i = 0; i < segment_.ty.size() - 1; ++i) {
135 for (size_t j1 = 0; j1 < segment_.xM[i].size(); ++j1) {
136 for (size_t j2 = 0; j2 < segment_.xM[i + 1].size(); ++j2) {
137 if (enhanced.xM[i][j1] < enhanced.xL[i + 1][j2] &&
138 enhanced.xU[i][j1] >= enhanced.xL[i + 1][j2]) {
139 enhanced.xU[i].push_back(enhanced.xL[i + 1][j2]);
140 enhanced.xL[i].push_back(enhanced.xL[i + 1][j2]);
141 enhanced.xM[i].push_back(enhanced.xL[i + 1][j2]);
142 } else if (enhanced.xM[i][j1] > enhanced.xU[i + 1][j2] &&
143 enhanced.xL[i][j1] <= enhanced.xU[i + 1][j2]) {
144 enhanced.xU[i].push_back(enhanced.xU[i + 1][j2]);
145 enhanced.xL[i].push_back(enhanced.xU[i + 1][j2]);
146 enhanced.xM[i].push_back(enhanced.xU[i + 1][j2]);
147 }
148
149 if (enhanced.xM[i + 1][j2] < enhanced.xL[i][j1] &&
150 enhanced.xU[i + 1][j2] >= enhanced.xL[i][j1]) {
151 enhanced.xU[i + 1].push_back(enhanced.xL[i][j1]);
152 enhanced.xL[i + 1].push_back(enhanced.xL[i][j1]);
153 enhanced.xM[i + 1].push_back(enhanced.xL[i][j1]);
154 } else if (enhanced.xM[i + 1][j2] > enhanced.xU[i][j1] &&
155 enhanced.xL[i + 1][j2] <= enhanced.xU[i][j1]) {
156 enhanced.xU[i + 1].push_back(enhanced.xU[i][j1]);
157 enhanced.xL[i + 1].push_back(enhanced.xU[i][j1]);
158 enhanced.xM[i + 1].push_back(enhanced.xU[i][j1]);
159 }
160 }
161 }
162
163 sort(enhanced.xL[i].begin(), enhanced.xL[i].end(),
164 [](double a, double b) { return a < b; });
165 sort(enhanced.xU[i].begin(), enhanced.xU[i].end(),
166 [](double a, double b) { return a < b; });
167 sort(enhanced.xM[i].begin(), enhanced.xM[i].end(),
168 [](double a, double b) { return a < b; });
169 }
170
171 segment_ = enhanced;
172}
