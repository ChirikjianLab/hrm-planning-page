74template <typename ObjectType>
75double computeObstacleMinSize(const std::vector<ObjectType>& obstacles) {
76 double minSizeObstacle = obstacles.at(0).getSemiAxis().at(0);
77 for (auto obs : obstacles) {
78 for (size_t i = 0; i < obs.getSemiAxis().size(); ++i) {
79 double size_obs = obs.getSemiAxis().at(i);
80 if (size_obs < minSizeObstacle) {
81 minSizeObstacle = size_obs;
82 }
83 }
84 }
85
86 return minSizeObstacle;
87}
